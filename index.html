from js import document, window from pyodide.ffi import create_proxy def generate_pdf(event): # Retrieve form values name = document.getElementById("name").value email = document.getElementById("email").value comments = document.getElementById("comments").value # Create a new PDF document pdf = window.jspdf.jsPDF() # Add text to the PDF pdf.text(10, 10, f"Name: {name}") pdf.text(10, 20, f"Email: {email}") pdf.text(10, 30, f"Comments: {comments}") # Save the PDF pdf.save("form.pdf") # Create a proxy for the generate_pdf function generate_pdf_proxy = create_proxy(generate_pdf) # Add event listener to the submit button document.getElementById("submitBtn").addEventListener("click", generate_pdf_proxy)
